```svelte
<script lang="ts">
    import { TelInput, countries } from 'svelte-tel-input';
    import type { DetailedValue, CountryCode } from 'svelte-tel-input/types';

    let selectedCountry: CountryCode | null = $state('HU');
    let value: string = $state('+36301234567');
    let valid: boolean = $state(true);
    let detailedValue: DetailedValue | null = $state(null);
</script>

<select bind:value={selectedCountry} aria-label="Country">
<option value={null}>Select country</option>
{#each countries as country (country.id)}
    <option value={country.iso2}>
    {country.name} (+{country.dialCode})
    </option>
{/each}
</select>

<TelInput
    bind:country={selectedCountry}
    bind:value
    bind:valid
    bind:detailedValue
    class="tel-input"
/>

{#if !valid}
    <p role="alert">Please enter a valid phone number.</p>
{/if}
```
